import pygame
from pygame.locals import *
import random

pygame.init()
running = True
size = width, height = (800, 800)
screen = pygame.display.set_mode(size)
pygame.display.set_caption("Car game")
screen.fill((60, 220, 0))
pygame.display.update()

while running:
    for event in pygame.event.get():
        if event.type == QUIT:
            running = False

pygame.quit()
