import pygame
from pygame.locals import *
import random

pygame.init()
running = True
screen = pygame.display.set_mode((800, 800))
pygame.display.set_caption("Car game")
screen.fill((60, 220, 0))
pygame.display.update()

while running:
    for event in pygame.event.get():
        if event.type == QUIT:
            running = False

pygame.quit()
